CREATE TABLE IF NOT EXISTS students (id INTEGER PRIMARY KEY, name TEXT NOT NULL, age INTEGER, grade TEXT);
CREATE TABLE IF NOT EXISTS contacts (id INTEGER PRIMARY KEY AUTOINCREMENT, name TEXT NOT NULL, email TEXT NOT NULL);
CREATE TABLE Item (code VARCHAR(50) PRIMARY KEY, quantity INT NOT NULL CHECK (quantity >= 0), price DECIMAL(10,2) NOT NULL CHECK (price >= 0));
CREATE TABLE Cart (cart_id INT PRIMARY KEY AUTO_INCREMENT); CREATE TABLE CartItem (cart_id INT NOT NULL, item_code VARCHAR(50) NOT NULL, quantity INT NOT NULL CHECK (quantity >= 0), price DECIMAL(10,2) NOT NULL CHECK (price >= 0), PRIMARY KEY (cart_id, item_code), FOREIGN KEY (cart_id) REFERENCES Cart(cart_id), FOREIGN KEY (item_code) REFERENCES Item(code));
